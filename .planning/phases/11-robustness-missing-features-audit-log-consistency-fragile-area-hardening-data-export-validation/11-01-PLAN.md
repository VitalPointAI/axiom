   ```python
   """Unified audit log writer.
   All mutation points import write_audit() to insert audit_log rows.
   Must be called within the caller's transaction boundary (same conn).
   """
   import json
   import logging
   logger = logging.getLogger(__name__)

   def write_audit(conn, *, user_id, entity_type, entity_id=None, action,
                   old_value=None, new_value, actor_type='system', notes=None):
       """Insert one audit_log row. Silently returns if conn is None (test compatibility)."""
       if conn is None:
           return
       try:
           cur = conn.cursor()
           cur.execute(
               """INSERT INTO audit_log
                  (user_id, entity_type, entity_id, action, old_value, new_value, actor_type, notes)
                  VALUES (%s, %s, %s, %s, %s, %s, %s, %s)""",
               (user_id, entity_type, entity_id, action,
                json.dumps(old_value) if old_value is not None else None,
                json.dumps(new_value) if not isinstance(new_value, str) else new_value,
                actor_type, notes)
           )
       except Exception:
           logger.warning("Failed to write audit log entry", exc_info=True)
   ```

The write_audit() helper silently returns when conn=None per RESEARCH.md guidance (Pitfall 5). This allows test callers that pass conn=None to skip audit writes. The try/except ensures audit log failures never crash the pipeline (consistent with "flag + continue" pattern).
