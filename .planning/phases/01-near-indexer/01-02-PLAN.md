```python
from db.init import get_connection
from indexers.nearblocks_client import NearBlocksClient
import time

def get_wallet_id(account_id):
    """Get or create wallet record."""
    conn = get_connection()
    row = conn.execute("SELECT id FROM wallets WHERE account_id = ?", (account_id,)).fetchone()
    if row:
        return row[0]
    conn.execute("INSERT INTO wallets (account_id) VALUES (?)", (account_id,))
    conn.commit()
    return conn.execute("SELECT id FROM wallets WHERE account_id = ?", (account_id,)).fetchone()[0]

def get_indexing_status(wallet_id):
    """Get current indexing progress."""
    conn = get_connection()
    row = conn.execute(
        "SELECT last_cursor, total_fetched, status FROM indexing_progress WHERE wallet_id = ?",
        (wallet_id,)
    ).fetchone()
    if row:
        return {"cursor": row[0], "fetched": row[1], "status": row[2]}
    return {"cursor": None, "fetched": 0, "status": "pending"}

def update_progress(wallet_id, cursor, fetched, status, error=None):
    """Update indexing progress."""
    conn = get_connection()
    conn.execute("""
        INSERT INTO indexing_progress (wallet_id, last_cursor, total_fetched, status, error_message, updated_at)
        VALUES (?, ?, ?, ?, ?, CURRENT_TIMESTAMP)
        ON CONFLICT(wallet_id) DO UPDATE SET
            last_cursor = excluded.last_cursor,
            total_fetched = excluded.total_fetched,
            status = excluded.status,
            error_message = excluded.error_message,
            updated_at = CURRENT_TIMESTAMP
    """, (wallet_id, cursor, fetched, status, error))
    conn.commit()

def index_account(account_id, force=False):
    """
    Index all transactions for an account.
    Resumable - saves cursor after each page.
    Returns: total transactions indexed
    """
    client = NearBlocksClient()
    wallet_id = get_wallet_id(account_id)
    
    # Check current status
    status = get_indexing_status(wallet_id)
    if status["status"] == "complete" and not force:
        print(f"{account_id}: Already complete ({status['fetched']} txs)")
        return status["fetched"]
    
    # Get total for progress display
    total_expected = client.get_transaction_count(account_id)
    print(f"{account_id}: {total_expected} total transactions")
    
    cursor = status["cursor"]
    fetched = status["fetched"]
    conn = get_connection()
    
    update_progress(wallet_id, cursor, fetched, "in_progress")
    
    try:
        while True:
            result = client.fetch_transactions(account_id, cursor=cursor, per_page=25)
            txns = result.get("txns", [])
            
            if not txns:
                break
            
            # Insert transactions
            for tx in txns:
                direction = "out" if tx.get("predecessor_account_id") == account_id else "in"
                counterparty = tx.get("receiver_account_id") if direction == "out" else tx.get("predecessor_account_id")
                
                actions = tx.get("actions", [])
                action_type = actions[0].get("action") if actions else None
                method_name = actions[0].get("method") if actions else None
                amount = str(tx.get("actions_agg", {}).get("deposit", 0))
                fee = str(tx.get("outcomes_agg", {}).get("transaction_fee", 0))
                
                try:
                    conn.execute("""
                        INSERT OR IGNORE INTO transactions 
                        (tx_hash, receipt_id, wallet_id, direction, counterparty, 
                         action_type, method_name, amount, fee, block_height, 
                         block_timestamp, success, raw_json)
                        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
                    """, (
                        tx.get("transaction_hash"),
                        tx.get("receipt_id"),
                        wallet_id,
                        direction,
                        counterparty,
                        action_type,
                        method_name,
                        amount,
                        fee,
                        tx.get("block", {}).get("block_height"),
                        tx.get("block_timestamp"),
                        tx.get("outcomes", {}).get("status", False),
                        str(tx)  # Store raw for debugging
                    ))
                except Exception as e:
                    print(f"  Warning: Error inserting tx {tx.get('transaction_hash')}: {e}")
            
            conn.commit()
            fetched += len(txns)
            cursor = result.get("cursor")
            
            # Save progress after each page
            update_progress(wallet_id, cursor, fetched, "in_progress")
            
            # Progress display
            pct = (fetched / total_expected * 100) if total_expected > 0 else 0
            print(f"  Progress: {fetched}/{total_expected} ({pct:.1f}%)")
            
            if not cursor:
                break
        
        update_progress(wallet_id, None, fetched, "complete")
        print(f"{account_id}: Complete! {fetched} transactions indexed")
        return fetched
        
    except Exception as e:
        update_progress(wallet_id, cursor, fetched, "error", str(e))
        print(f"{account_id}: Error at {fetched} txs - {e}")
        raise
```

Create CLI script for batch indexing with progress tracking:

```python
#!/usr/bin/env python3
import argparse
import sys
import time
from db.init import get_connection
from indexers.near_indexer import index_account, get_wallet_id, get_indexing_status

def get_all_wallets():
    """Get all wallets ordered by account_id."""
    conn = get_connection()
    return [row[0] for row in conn.execute(
        "SELECT account_id FROM wallets ORDER BY account_id"
    ).fetchall()]

def get_status_summary():
    """Get indexing status summary."""
    conn = get_connection()
    rows = conn.execute("""
        SELECT status, COUNT(*) 
        FROM indexing_progress 
        GROUP BY status
    """).fetchall()
    return dict(rows)

def main():
    parser = argparse.ArgumentParser(description="Index NEAR transactions")
    parser.add_argument("--account", help="Index single account")
    parser.add_argument("--status", action="store_true", help="Show status only")
    parser.add_argument("--force", action="store_true", help="Re-index complete accounts")
    parser.add_argument("--limit", type=int, help="Max accounts to process")
    args = parser.parse_args()
    
    if args.status:
        summary = get_status_summary()
        print("Indexing Status:")
        for status, count in summary.items():
            print(f"  {status}: {count}")
        return
    
    if args.account:
        index_account(args.account, force=args.force)
        return
    
    # Batch mode - index all pending/error accounts
    wallets = get_all_wallets()
    processed = 0
    
    for account_id in wallets:
        wallet_id = get_wallet_id(account_id)
        status = get_indexing_status(wallet_id)
        
        if status["status"] == "complete" and not args.force:
            continue
        
        print(f"\n{'='*60}")
        print(f"Processing: {account_id}")
        print(f"{'='*60}")
        
        try:
            index_account(account_id, force=args.force)
            processed += 1
        except KeyboardInterrupt:
            print("\nInterrupted! Progress saved. Run again to resume.")
            sys.exit(0)
        except Exception as e:
            print(f"Error: {e}")
            continue
        
        if args.limit and processed >= args.limit:
            print(f"\nLimit reached ({args.limit} accounts)")
            break
    
    print(f"\nProcessed {processed} accounts")
    summary = get_status_summary()
    print("Final status:", summary)

if __name__ == "__main__":
    main()
```

Usage:
- `python scripts/index_all.py --status` - Show progress
- `python scripts/index_all.py --account vitalpointai.near` - Single account
- `python scripts/index_all.py --limit 5` - First 5 pending accounts
- `python scripts/index_all.py` - All pending accounts
- Ctrl+C to interrupt (progress saved)

```python
import requests
from db.init import get_connection
from indexers.near_indexer import get_wallet_id

FASTNEAR_RPC = "https://free.rpc.fastnear.com"

def get_onchain_balance(account_id):
    """Get current NEAR balance via FastNear RPC."""
    response = requests.post(
        FASTNEAR_RPC,
        json={
            "jsonrpc": "2.0",
            "id": "1",
            "method": "query",
            "params": {
                "request_type": "view_account",
                "finality": "final",
                "account_id": account_id
            }
        },
        timeout=10
    )
    result = response.json().get("result", {})
    amount = int(result.get("amount", 0))
    return amount / 1e24  # Convert yoctoNEAR to NEAR

def calculate_balance_from_txs(account_id):
    """
    Calculate balance from indexed transactions.
    Note: This is approximate - doesn't account for staking, storage, etc.
    """
    conn = get_connection()
    wallet_id = get_wallet_id(account_id)
    
    # Sum incoming transfers
    incoming = conn.execute("""
        SELECT COALESCE(SUM(CAST(amount AS REAL)), 0) 
        FROM transactions 
        WHERE wallet_id = ? AND direction = 'in' AND action_type = 'TRANSFER'
    """, (wallet_id,)).fetchone()[0]
    
    # Sum outgoing transfers + fees
    outgoing = conn.execute("""
        SELECT COALESCE(SUM(CAST(amount AS REAL) + CAST(fee AS REAL)), 0)
        FROM transactions 
        WHERE wallet_id = ? AND direction = 'out'
    """, (wallet_id,)).fetchone()[0]
    
    # Convert from yoctoNEAR
    incoming_near = incoming / 1e24 if incoming else 0
    outgoing_near = outgoing / 1e24 if outgoing else 0
    
    return incoming_near - outgoing_near

def reconcile_account(account_id):
    """
    Compare calculated vs on-chain balance.
    Returns dict with comparison data.
    """
    onchain = get_onchain_balance(account_id)
    calculated = calculate_balance_from_txs(account_id)
    difference = abs(onchain - calculated)
    
    # For now, just show comparison (won't match due to staking, etc)
    return {
        "account_id": account_id,
        "onchain": onchain,
        "calculated": calculated,
        "difference": difference,
        "note": "Difference expected - staking/storage not yet accounted for"
    }

if __name__ == "__main__":
    import sys
    account = sys.argv[1] if len(sys.argv) > 1 else "vitalpointai.near"
    result = reconcile_account(account)
    print(f"Account: {result['account_id']}")
    print(f"On-chain: {result['onchain']:.4f} NEAR")
    print(f"Calculated: {result['calculated']:.4f} NEAR")
    print(f"Difference: {result['difference']:.4f} NEAR")
    print(f"Note: {result['note']}")
```
