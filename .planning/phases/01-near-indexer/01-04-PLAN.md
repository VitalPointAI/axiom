   ```typescript

   const walletId = walletResult.id;
   const userId = user.id;

   // Job 1: Full transaction history sync
   await db.run(
     `INSERT INTO indexing_jobs (user_id, wallet_id, job_type, chain, status, priority)
      VALUES ($1, $2, 'full_sync', $3, 'queued', 10)`,
     [userId, walletId, chain || 'near']
   );

   // Job 2: Staking rewards sync
   await db.run(
     `INSERT INTO indexing_jobs (user_id, wallet_id, job_type, chain, status, priority)
      VALUES ($1, $2, 'staking_sync', $3, 'queued', 5)`,
     [userId, walletId, chain || 'near']
   );

   // Job 3: Lockup contract sync (only for NEAR chain)
   if (!chain || chain === 'near') {
     await db.run(
       `INSERT INTO indexing_jobs (user_id, wallet_id, job_type, chain, status, priority)
        VALUES ($1, $2, 'lockup_sync', 'near', 'queued', 3)`,
       [userId, walletId]
     );
   }
   ```
