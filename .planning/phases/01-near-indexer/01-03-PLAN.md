Create staking rewards tracker using NearBlocks + FastNear RPC:

```python
import requests
from indexers.nearblocks_client import NearBlocksClient

FASTNEAR_RPC = "https://free.rpc.fastnear.com"

def get_pool_balance(account_id, pool_id):
    """Query validator pool for account's current staked balance."""
    try:
        response = requests.post(
            FASTNEAR_RPC,
            json={
                "jsonrpc": "2.0",
                "id": "1",
                "method": "query",
                "params": {
                    "request_type": "call_function",
                    "finality": "final",
                    "account_id": pool_id,
                    "method_name": "get_account_staked_balance",
                    "args_base64": __import__('base64').b64encode(
                        f'{{"account_id": "{account_id}"}}'.encode()
                    ).decode()
                }
            },
            timeout=10
        )
        result = response.json().get("result", {})
        if "result" in result:
            # Decode the result (it's a JSON string in bytes)
            import json
            balance_str = bytes(result["result"]).decode()
            return int(balance_str.strip('"')) / 1e24
    except Exception as e:
        print(f"  Warning: Could not query {pool_id}: {e}")
    return 0

def get_staking_summary(account_id):
    """
    Get staking summary using NearBlocks kitwallet endpoint.
    Returns deposits/withdrawals per validator.
    """
    client = NearBlocksClient()
    deposits = client.fetch_staking_deposits(account_id)
    
    summary = []
    for item in deposits:
        validator_id = item.get("validator_id")
        deposit = int(item.get("deposit", 0))
        
        # Negative deposit means withdrawn, positive means currently staked
        if deposit > 0:
            # Get current balance from pool
            current = get_pool_balance(account_id, validator_id)
            deposited = deposit / 1e24  # NearBlocks gives in yoctoNEAR
            rewards = current - deposited if current > 0 else 0
        else:
            # Already withdrawn - deposit shows net (deposited - withdrawn)
            current = 0
            deposited = 0
            rewards = 0  # Can't calculate rewards for closed positions
        
        summary.append({
            "validator_id": validator_id,
            "net_deposit_yocto": deposit,
            "current_staked": current,
            "estimated_rewards": rewards
        })
    
    return summary

def calculate_rewards(account_id):
    """
    Calculate total staking rewards.
    
    Formula: rewards = current_staked - total_deposited + total_withdrawn
    
    Note: For closed positions (fully withdrawn), we can't calculate rewards
    accurately without full transaction history. Mark as "unknown".
    """
    summary = get_staking_summary(account_id)
    
    total_current = 0
    total_rewards = 0
    validators_with_rewards = []
    validators_unknown = []
    
    for v in summary:
        if v["current_staked"] > 0:
            total_current += v["current_staked"]
            total_rewards += v["estimated_rewards"]
            if v["estimated_rewards"] > 0:
                validators_with_rewards.append(v)
        elif v["net_deposit_yocto"] < 0:
            # Withdrawn validator - rewards unknown without full history
            validators_unknown.append(v["validator_id"])
    
    return {
        "account_id": account_id,
        "total_currently_staked": total_current,
        "total_estimated_rewards": total_rewards,
        "validators_with_current_stake": validators_with_rewards,
        "validators_fully_withdrawn": validators_unknown,
        "note": "Rewards for withdrawn validators require full tx history analysis"
    }

if __name__ == "__main__":
    import sys
    import json
    account = sys.argv[1] if len(sys.argv) > 1 else "vitalpointai.near"
    result = calculate_rewards(account)
    print(json.dumps(result, indent=2))
```

Key approach:
- Use NearBlocks kitwallet endpoint for deposit summary (1 API call)
- Query each active validator pool for current balance
- Calculate rewards = current_staked - net_deposits
- Flag withdrawn validators as "unknown" (need tx history)

```python
import requests
from indexers.nearblocks_client import NearBlocksClient

FASTNEAR_RPC = "https://free.rpc.fastnear.com"
LOCKUP_ACCOUNT = "db59d3239f2939bb7d8a4a578aceaa8c85ee8e3f.lockup.near"

def get_lockup_state(lockup_account=LOCKUP_ACCOUNT):
    """Query lockup contract state via RPC."""
    try:
        response = requests.post(
            FASTNEAR_RPC,
            json={
                "jsonrpc": "2.0",
                "id": "1",
                "method": "query",
                "params": {
                    "request_type": "view_account",
                    "finality": "final",
                    "account_id": lockup_account
                }
            },
            timeout=10
        )
        result = response.json().get("result", {})
        return {
            "balance": int(result.get("amount", 0)) / 1e24,
            "storage_used": result.get("storage_usage", 0)
        }
    except Exception as e:
        return {"error": str(e)}

def get_lockup_info(lockup_account=LOCKUP_ACCOUNT):
    """Get lockup contract info via contract call."""
    methods = [
        "get_balance",
        "get_locked_amount", 
        "get_owners_balance",
        "get_liquid_owners_balance"
    ]
    
    info = {}
    for method in methods:
        try:
            response = requests.post(
                FASTNEAR_RPC,
                json={
                    "jsonrpc": "2.0",
                    "id": "1",
                    "method": "query",
                    "params": {
                        "request_type": "call_function",
                        "finality": "final",
                        "account_id": lockup_account,
                        "method_name": method,
                        "args_base64": "e30="  # {} in base64
                    }
                },
                timeout=10
            )
            result = response.json().get("result", {})
            if "result" in result:
                value_bytes = bytes(result["result"])
                value_str = value_bytes.decode().strip('"')
                info[method] = int(value_str) / 1e24 if value_str.isdigit() else value_str
        except:
            pass
    
    return info

def get_lockup_summary():
    """
    Get lockup contract summary.
    
    Aaron confirmed: Vesting COMPLETE as of ~2021 (1 year after opening).
    This is historical data for tax records.
    """
    state = get_lockup_state()
    info = get_lockup_info()
    
    # Get transaction history for the lockup
    client = NearBlocksClient()
    tx_count = client.get_transaction_count(LOCKUP_ACCOUNT)
    
    return {
        "lockup_account": LOCKUP_ACCOUNT,
        "owner": "vitalpointai.near",
        "vesting_status": "COMPLETE (as of ~2021)",
        "current_balance": state.get("balance", 0),
        "contract_info": info,
        "transaction_count": tx_count,
        "note": "Lockup vesting complete. All funds are liquid and have been transferred/staked.",
        "tax_treatment": "Tokens became taxable income when vesting unlocked (2020-2021)"
    }

if __name__ == "__main__":
    import json
    result = get_lockup_summary()
    print(json.dumps(result, indent=2))
```
