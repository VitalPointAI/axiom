```python
def _get_indexed_blocks(self, account_id: str) -> Optional[list]:
    """Query account_block_index_v2 via dictionary for segment-based block ranges.

    Returns a sorted list of block heights (expanded from segments),
    or None if the index table is empty / not yet built.
    """
    conn = self.db_pool.getconn()
    try:
        cur = conn.cursor()
        # Check if the index has been built
        try:
            cur.execute("SELECT last_processed_block FROM account_indexer_state WHERE id = 1")
            state = cur.fetchone()
            if not state or state[0] < 1_000_000:
                cur.close()
                return None
        except Exception:
            cur.close()
            conn.rollback()
            return None

        # Resolve account string to integer via dictionary
        cur.execute(
            "SELECT id FROM account_dictionary WHERE account_id = %s",
            (account_id.lower(),),
        )
        dict_row = cur.fetchone()
        if not dict_row:
            # Account not in dictionary — check if index is near chain tip
            # before concluding the account has no activity
            cur.execute("SELECT last_processed_block FROM account_indexer_state WHERE id = 1")
            last_block = cur.fetchone()[0]
            cur.close()

            try:
                tip = self.client.get_final_block_height()
                if tip and last_block >= tip - 1000:
                    return []  # Index is current, account genuinely has no activity
            except Exception:
                pass
            return None  # Index might not be current, fall back to full scan

        account_int = dict_row[0]

        # Query v2 table for segment_start values
        cur.execute(
            "SELECT segment_start FROM account_block_index_v2 WHERE account_int = %s ORDER BY segment_start",
            (account_int,),
        )
        rows = cur.fetchall()
        cur.close()

        if not rows:
            return []

        # Expand segments to block ranges
        # Each segment_start represents a 1000-block range: [segment_start, segment_start + 999]
        # Return the list of segment_start values — the caller (_sync_from_index)
        # will scan each 1000-block range via neardata.xyz
        segments = [r[0] for r in rows]
        return segments
    finally:
        self.db_pool.putconn(conn)
```

Create `scripts/migrate_to_v2.py` — a standalone Python script that migrates existing `account_block_index` data into the new `account_dictionary` + `account_block_index_v2` tables.

The script has two phases:

**Phase 1: Populate account_dictionary**
```python
def populate_dictionary(conn):
    """Insert all distinct account_ids from old table into account_dictionary."""
    cur = conn.cursor()
    logger.info("Phase 1: Populating account_dictionary from existing index...")
    cur.execute("""
        INSERT INTO account_dictionary (account_id)
        SELECT DISTINCT account_id FROM account_block_index
        ON CONFLICT (account_id) DO NOTHING
    """)
    conn.commit()
    count = cur.rowcount
    cur.close()
    logger.info("Inserted %d accounts into dictionary.", count)
    return count
```

**Phase 2: Copy rows to v2 in batches**
Process in block_height range batches of 1,000,000 blocks (1M per batch). For each batch:
```python
def migrate_batch(conn, start_block, end_block):
    """Copy one block range from old table to v2 via dictionary join + segment calculation."""
    cur = conn.cursor()
    cur.execute("""
        INSERT INTO account_block_index_v2 (account_int, segment_start)
        SELECT DISTINCT d.id, ((abi.block_height / 1000) * 1000)::integer
        FROM account_block_index abi
        JOIN account_dictionary d ON d.account_id = abi.account_id
        WHERE abi.block_height >= %s AND abi.block_height < %s
        ON CONFLICT DO NOTHING
    """, (start_block, end_block))
    inserted = cur.rowcount
    conn.commit()
    cur.close()
    return inserted
```

Note the segment calculation in SQL: `((abi.block_height / 1000) * 1000)::integer`. PostgreSQL integer division truncates, so `block_height / 1000` already floors. Multiply by 1000 to get the segment start. Cast to integer since block_height is BIGINT in the old table.

The DISTINCT clause is critical: multiple rows with block heights in the same segment collapse to one row per (account_int, segment_start) pair. This is where the segment-based row count reduction comes from.

**Main function:**
```python
def main():
    import psycopg2
    database_url = os.environ.get("DATABASE_URL", "postgresql://neartax:neartax@127.0.0.1:5433/neartax")
    conn = psycopg2.connect(database_url)

    # Phase 1: Dictionary
    populate_dictionary(conn)

    # Get block range from account_indexer_state
    cur = conn.cursor()
    cur.execute("SELECT last_processed_block FROM account_indexer_state WHERE id = 1")
    max_block = cur.fetchone()[0]
    cur.close()

    # Phase 2: Batch migration
    BATCH_SIZE = 1_000_000
    GENESIS = 9_820_210
    total_inserted = 0
    start = GENESIS

    while start < max_block:
        end = min(start + BATCH_SIZE, max_block + 1)
        t0 = time.monotonic()
        inserted = migrate_batch(conn, start, end)
        elapsed = time.monotonic() - t0
        total_inserted += inserted
        pct = (end - GENESIS) / (max_block - GENESIS) * 100
        logger.info(
            "Batch %d-%d: %d rows (%.1fs) | Total: %d | %.1f%%",
            start, end, inserted, elapsed, total_inserted, pct,
        )
        start = end

    logger.info("Migration complete. Total rows in v2: %d", total_inserted)

    # Verify
    cur = conn.cursor()
    cur.execute("SELECT COUNT(*) FROM account_dictionary")
    dict_count = cur.fetchone()[0]
    cur.execute("SELECT reltuples::bigint FROM pg_class WHERE relname = 'account_block_index_v2'")
    v2_estimate = cur.fetchone()[0]
    cur.close()
    logger.info("Verification: %d dictionary entries, ~%d v2 rows", dict_count, v2_estimate)

    conn.close()
```
