```python
"""Seeds classification_rules table from hardcoded patterns.

Converts the rule-of-thumb logic from tax/categories.py into database records
so the classifier engine can load rules dynamically without code changes.
Run once during migration, or idempotently via seed_classification_rules().
"""

def get_near_rules() -> list[dict]:
    """Return NEAR classification rules derived from tax/categories.py patterns.

    Each rule is a dict: {name, chain, pattern (JSONB), category, confidence, priority}

    Rules to port (from categorize_near_transaction):
    1. Staking deposits: method_name IN (deposit_and_stake, stake, deposit) + counterparty_suffix .poolv1.near/.pool.near -> STAKE, 0.95
    2. Unstaking: method_name IN (unstake, unstake_all) + pool suffix -> UNSTAKE, 0.95
    3. Staking withdrawal: method_name IN (withdraw, withdraw_all) + pool suffix -> UNSTAKE, 0.90
    4. Validator ping: method_name=ping + pool suffix -> INTERNAL, 0.99
    5. Staking reward: direction=in + amount>0 + pool suffix -> REWARD, 0.85
    6. DEX swap out: method_name IN (swap, ft_transfer_call) + DEX contract + direction=out -> TRADE, 0.90
    7. DEX swap in: DEX contract + direction=in + amount>0 -> TRADE, 0.85
    8. LP add: method_name IN (add_liquidity, add_stable_liquidity) + DEX -> LIQUIDITY_IN, 0.90
    9. LP remove: method_name=remove_liquidity + DEX -> LIQUIDITY_OUT, 0.90
    10. Lending deposit: method_name IN (supply, deposit) + Burrow -> COLLATERAL_IN, 0.90
    11. Lending withdrawal: method_name=withdraw + Burrow -> COLLATERAL_OUT, 0.90
    12. Borrow: method_name=borrow + Burrow -> LOAN_BORROW, 0.90
    13. Repay: method_name=repay + Burrow -> LOAN_REPAY, 0.90
    14. Lending interest: claim_reward or reward in method + Burrow -> INTEREST, 0.85
    15. FT transfer own wallet: ft_transfer/ft_transfer_call + is_own_wallet -> TRANSFER_IN/OUT, 0.90
    16. NFT mint: nft_mint/nft_mint_batch -> NFT_MINT, 0.95
    17. Airdrop claim: claim/claim_airdrop/claim_tokens -> AIRDROP, 0.85
    18. Basic NEAR transfer: action_type=TRANSFER -> DEPOSIT/WITHDRAWAL, 0.70
    19. Account operations: CREATE_ACCOUNT -> ACCOUNT_CREATE, ADD_KEY/DELETE_KEY -> INTERNAL
    20. Storage: storage_deposit/storage_withdraw -> FEE, 0.90
    """

def get_evm_rules() -> list[dict]:
    """Return EVM classification rules.

    1. DEX swap signatures (all 10 from EVMDecoder.DEX_SIGNATURES) -> TRADE, 0.90
    2. Lending signatures (Aave V2) -> appropriate category, 0.85
    3. LP signatures -> LIQUIDITY_IN/OUT, 0.85
    4. Plain ETH/token transfer -> DEPOSIT/WITHDRAWAL, 0.70
    """

def get_exchange_rules() -> list[dict]:
    """Return exchange transaction classification rules.

    Port from classify_exchange_transaction() in engine/classifier.py:
    1. tx_type buy/purchase -> BUY, 0.95
    2. tx_type sell/sale -> SELL, 0.95
    3. tx_type swap/convert/trade -> TRADE, 0.95
    4. tx_type staking_reward/interest/reward/dividend -> REWARD/INTEREST/INCOME, 0.95
    5. tx_type send/withdrawal -> WITHDRAWAL, 0.80
    6. tx_type receive/deposit -> DEPOSIT, 0.80
    7. tx_type airdrop -> AIRDROP, 0.90
    """

def seed_classification_rules(pool) -> int:
    """Insert all rules into classification_rules table. Idempotent via ON CONFLICT.

    Uses INSERT ... ON CONFLICT (name) DO UPDATE to allow re-running.
    The classification_rules table has a UNIQUE constraint on name (uq_cr_name)
    created by migration 003, which makes ON CONFLICT (name) work.
    Returns count of rules inserted/updated.
    """
    rules = get_near_rules() + get_evm_rules() + get_exchange_rules()
    conn = pool.getconn()
    try:
        cur = conn.cursor()
        count = 0
        for rule in rules:
            cur.execute(
                """INSERT INTO classification_rules
                   (name, chain, pattern, category, confidence, priority, is_active, created_at, updated_at)
                   VALUES (%s, %s, %s::jsonb, %s, %s, %s, TRUE, NOW(), NOW())
                   ON CONFLICT (name) DO UPDATE SET
                     pattern = EXCLUDED.pattern,
                     category = EXCLUDED.category,
                     confidence = EXCLUDED.confidence,
                     priority = EXCLUDED.priority,
                     updated_at = NOW()
                """,
                (rule["name"], rule["chain"], json.dumps(rule["pattern"]),
                 rule["category"], rule["confidence"], rule["priority"]),
            )
            count += 1
        conn.commit()
        return count
    finally:
        pool.putconn(conn)
```

Priority ordering: higher priority = checked first. Set staking rules at priority 100, DEX at 90, lending at 80, basic transfers at 50, fallbacks at 10.
